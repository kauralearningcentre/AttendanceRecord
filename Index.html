Google Sheets & Apps Script Integration

Connect this Web App directly to your Google Spreadsheet for two-way sync!

Apps Script Code (Code.gs)

// Google Apps Script for KLC Attendance System 2026-2027
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var studentSheet = ss.getSheetByName("Students") || ss.getActiveSheet();
  var logSheet = ss.getSheetByName("Logs") || ss.insertSheet("Logs");
  
  var studentRows = studentSheet.getDataRange().getValues();
  var students = [];
  for (var i = 1; i < studentRows.length; i++) {
    if (studentRows[i][0]) {
      students.push({
        id: studentRows[i][0],
        name: studentRows[i][1],
        classes: String(studentRows[i][2]).split(",").map(function(c) { return c.trim(); }),
        status: studentRows[i][3] || "Active"
      });
    }
  }
  
  var logRows = logSheet.getDataRange().getValues();
  var logs = [];
  for (var j = 1; j < logRows.length; j++) {
    if (logRows[j][0]) {
      logs.push({
        timestamp: logRows[j][0],
        tanggal: logRows[j][1],
        idSiswa: logRows[j][2],
        namaSiswa: logRows[j][3],
        kodeKelas: logRows[j][4],
        namaKelas: logRows[j][5],
        status: logRows[j][6],
        materi: logRows[j][7],
        teacherNotes: logRows[j][8],
        remarks: logRows[j][9]
      });
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    students: students,
    logs: logs
  })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = ss.getSheetByName("Logs") || ss.insertSheet("Logs");
  
  if (data.action === "saveAttendance") {
    var entries = data.data;
    for (var i = 0; i < entries.length; i++) {
      var row = entries[i];
      logSheet.appendRow([
        row.timestamp, row.tanggal, row.idSiswa, row.namaSiswa,
        row.kodeKelas, row.namaKelas, row.status, row.materi || "",
        row.teacherNotes || "", row.remarks || ""
      ]);
    }
  }
  return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON);
}
