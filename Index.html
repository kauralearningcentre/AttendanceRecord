function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  
  try {
    if (!e || !e.postData || !e.postData.contents) {
      throw new Error("No POST data payload provided.");
    }

    var data = JSON.parse(e.postData.contents);
    var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
    
    function getOrCreateSheet(name, headers) {
      var sh = spreadsheet.getSheetByName(name);
      if (!sh) {
        sh = spreadsheet.insertSheet(name);
        if (headers && headers.length > 0) {
          sh.appendRow(headers);
        }
      }
      return sh;
    }
    
    if (data.action === 'addStudent') {
      var sheet = getOrCreateSheet('Student_Data', ['ID Siswa', 'Full Name', 'Enrolled Classes', 'Class Name', 'Class ID', 'Status']);
      var rows = sheet.getDataRange().getValues();
      var targetId = String(data.nis || '').trim().toUpperCase();
      var updated = false;
      
      for (var i = 1; i < rows.length; i++) {
        if (String(rows[i][0]).trim().toUpperCase() === targetId) {
          sheet.getRange(i + 1, 1, 1, 6).setValues([[
            data.nis, 
            data.name, 
            data.enrolledClasses || '', 
            data.className || '', 
            data.classId || '', 
            data.status || 'Active'
          ]]);
          updated = true;
          break;
        }
      }
      if (!updated) {
        sheet.appendRow([data.nis, data.name, data.enrolledClasses || '', data.className || '', data.classId || '', data.status || 'Active']);
      }
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    if (data.action === 'addClass') {
      var classSheet = getOrCreateSheet('KLC_Classes', ['Class ID', 'Class Name', 'Category', 'Schedule', 'Room']);
      var cRows = classSheet.getDataRange().getValues();
      var targetClassId = String(data.classId || '').trim().toUpperCase();
      var classUpdated = false;
      
      for (var c = 1; c < cRows.length; c++) {
        if (String(cRows[c][0]).trim().toUpperCase() === targetClassId) {
          classSheet.getRange(c + 1, 1, 1, 5).setValues([[
            data.classId,
            data.className,
            data.category || '',
            data.schedule || '',
            data.room || 'Community Hall Kaura Manggis'
          ]]);
          classUpdated = true;
          break;
        }
      }
      if (!classUpdated) {
        classSheet.appendRow([data.classId, data.className, data.category || '', data.schedule || '', data.room || 'Community Hall Kaura Manggis']);
      }
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    if (data.action === 'deleteStudent') {
      var sheet = spreadsheet.getSheetByName('Student_Data');
      if (sheet) {
        var rows = sheet.getDataRange().getValues();
        var targetId = String(data.nis || '').trim().toUpperCase();
        
        for (var i = rows.length - 1; i >= 1; i--) {
          if (String(rows[i][0]).trim().toUpperCase() === targetId) {
            sheet.deleteRow(i + 1);
            break;
          }
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    var logHeaders = ['Timestamp', 'Date', 'Student ID', 'Student Name', 'Class ID', 'Class Name', 'Status', 'Topic', 'Teacher Note', 'Module Link', 'Remark'];
    var logSheet = getOrCreateSheet('Attendance_Log', logHeaders);
    var moduleSheet = getOrCreateSheet('Learning_Modules', ['Module ID', 'Class ID', 'Class Name', 'Topic', 'Module Link', 'Description', 'Date Added']);
    
    var records = data.records || data.data;
    
    if (Array.isArray(records) && records.length > 0) {
      var sampleRecord = records[0];
      var targetClass = String(sampleRecord.kodeKelas || sampleRecord.classId || '').trim().toUpperCase();
      
      var rawTargetDate = String(sampleRecord.tanggal || sampleRecord.date || '').trim();
      var targetDate = rawTargetDate.replace(/\//g, '-');
      if (targetDate.includes('T')) targetDate = targetDate.split('T')[0];
      if (targetDate.includes(' ')) targetDate = targetDate.split(' ')[0];

      if (targetClass && targetDate) {
        var existingLogRows = logSheet.getDataRange().getValues();
        for (var l = existingLogRows.length - 1; l >= 1; l--) {
          var rawRowDate = existingLogRows[l][1];
          var rowDate = '';
          
          if (rawRowDate instanceof Date) {
            rowDate = Utilities.formatDate(rawRowDate, spreadsheet.getSpreadsheetTimeZone(), "yyyy-MM-dd");
          } else {
            rowDate = String(rawRowDate || '').trim().replace(/\//g, '-');
            if (rowDate.includes('T')) rowDate = rowDate.split('T')[0];
            if (rowDate.includes(' ')) rowDate = rowDate.split(' ')[0];
          }

          var rowClass = String(existingLogRows[l][4] || '').trim().toUpperCase();
          if (rowDate === targetDate && rowClass === targetClass) {
            logSheet.deleteRow(l + 1);
          }
        }
      }
      
      records.forEach(function(r) {
        logSheet.appendRow([
          r.timestamp || new Date(),
          r.tanggal || r.date,
          r.idSiswa || r.studentId,
          r.namaSiswa || r.studentName,
          r.kodeKelas || r.classId,
          r.namaKelas || r.className,
          r.status,
          r.materi || r.topic,
          r.teacherNotes || r.teacherNote || '', 
          r.moduleLink || r.modulLink || r.materiLink || r.link || '', 
          r.remarks || r.remark || ''              
        ]);
      });

      var moduleLink = sampleRecord.moduleLink || sampleRecord.modulLink || sampleRecord.materiLink || sampleRecord.link || '';
      var topic = sampleRecord.materi || sampleRecord.topic || '';
      var className = sampleRecord.namaKelas || sampleRecord.className || '';

      if (moduleLink && moduleLink.trim() !== '') {
        var mRows = moduleSheet.getDataRange().getValues();
        var moduleUpdated = false;
        var cleanLink = moduleLink.trim();

        for (var m = 1; m < mRows.length; m++) {
          var existingLink = String(mRows[m][4] || '').trim();
          var existingClass = String(mRows[m][1] || '').trim().toUpperCase();
          var existingTopic = String(mRows[m][3] || '').trim().toLowerCase();

          if (existingLink === cleanLink || (existingClass === targetClass && existingTopic === topic.toLowerCase())) {
            moduleSheet.getRange(m + 1, 1, 1, 7).setValues([[
              mRows[m][0] || 'MOD-' + Date.now(),
              targetClass,
              className,
              topic,
              cleanLink,
              sampleRecord.teacherNotes || sampleRecord.description || 'Module from class session',
              targetDate || Utilities.formatDate(new Date(), spreadsheet.getSpreadsheetTimeZone(), "yyyy-MM-dd")
            ]]);
            moduleUpdated = true;
            break;
          }
        }

        if (!moduleUpdated) {
          var newModuleId = 'MOD-' + Date.now().toString().slice(-6);
          moduleSheet.appendRow([
            newModuleId,
            targetClass,
            className,
            topic,
            cleanLink,
            sampleRecord.teacherNotes || sampleRecord.description || 'Module from class session',
            targetDate || Utilities.formatDate(new Date(), spreadsheet.getSpreadsheetTimeZone(), "yyyy-MM-dd")
          ]);
        }
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
  finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var tz = ss.getSpreadsheetTimeZone();
  
  var classes = [];
  var classSheet = ss.getSheetByName("KLC_Classes");
  if (classSheet) {
    var cRows = classSheet.getDataRange().getValues();
    for (var c = 1; c < cRows.length; c++) {
      var cr = cRows[c];
      if (!cr[0]) continue;
      classes.push({
        id: String(cr[0]).trim(),
        name: cr[1] || cr[0],
        category: cr[2] || (String(cr[0]).toUpperCase().includes('ENG') ? 'English' : 'Computer'),
        schedule: cr[3] || 'Scheduled Session',
        room: cr[4] || 'Community Hall Kaura Manggis'
      });
    }
  }

  var studentSheet = ss.getSheetByName("Student_Data");
  var students = [];
  var classMap = {};
  
  if (studentSheet) {
    var rows = studentSheet.getDataRange().getValues();
    for (var i = 1; i < rows.length; i++) {
      var row = rows[i];
      var idSiswa = row[0];
      if (!idSiswa) continue;
      
      var fullName = row[1];
      var enrolledClasses = row[2];
      var className = row[3];
      var classId = row[4];
      var status = row[5] || 'Active';
      
      students.push({
        id: String(idSiswa).trim(),
        name: fullName,
        fullName: fullName,
        enrolledClasses: enrolledClasses,
        className: className,
        classId: classId,
        classes: enrolledClasses ? String(enrolledClasses).split(',').map(function(c) { return c.trim(); }) : (classId ? [classId] : []),
        status: status
      });
      
      if (classId && !classMap[classId]) {
        classMap[classId] = {
          id: String(classId).trim(),
          name: className || classId,
          category: String(classId).toUpperCase().includes('ENG') ? 'English' : 'Computer',
          schedule: 'Scheduled Session',
          room: 'Community Hall Kaura Manggis'
        };
      }
    }
  }

  if (classes.length === 0) {
    for (var key in classMap) {
      classes.push(classMap[key]);
    }
  }

  var logSheet = ss.getSheetByName("Attendance_Log");
  var logs = [];
  if (logSheet) {
    var logRows = logSheet.getDataRange().getValues();
    for (var j = 1; j < logRows.length; j++) {
      var lr = logRows[j];
      if (!lr[2]) continue;
      
      var rawDate = lr[1];
      var cleanDateStr = '';
      if (rawDate instanceof Date) {
        cleanDateStr = Utilities.formatDate(rawDate, tz, "yyyy-MM-dd");
      } else {
        var strVal = String(rawDate || '').trim().replace(/\//g, '-');
        if (strVal.includes('T')) {
          cleanDateStr = strVal.split('T')[0];
        } else if (strVal.includes(' ')) {
          cleanDateStr = strVal.split(' ')[0];
        } else {
          cleanDateStr = strVal;
        }
      }

      logs.push({
        timestamp: lr[0],       
        tanggal: cleanDateStr,  
        idSiswa: String(lr[2]).trim(), 
        namaSiswa: lr[3],       
        kodeKelas: lr[4],       
        namaKelas: lr[5],       
        status: lr[6],          
        materi: lr[7],          
        teacherNotes: lr[8],    
        moduleLink: lr[9],      
        remarks: lr[10]         
      });
    }
  }

  var moduleSheet = ss.getSheetByName("Learning_Modules");
  var modules = [];
  if (moduleSheet) {
    var mRows = moduleSheet.getDataRange().getValues();
    for (var k = 1; k < mRows.length; k++) {
      var mr = mRows[k];
      if (!mr[4]) continue;
      
      var rawAddDate = mr[6];
      var cleanAddDate = '';
      if (rawAddDate instanceof Date) {
        cleanAddDate = Utilities.formatDate(rawAddDate, tz, "yyyy-MM-dd");
      } else {
        cleanAddDate = String(rawAddDate || '').trim();
      }

      modules.push({
        id: mr[0] || ('MOD-' + k),
        classId: mr[1],
        className: mr[2],
        topic: mr[3],
        moduleLink: mr[4],
        description: mr[5],
        dateAdded: cleanAddDate
      });
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    students: students,
    classes: classes,
    logs: logs,
    modules: modules
  })).setMimeType(ContentService.MimeType.JSON);
}
