Integration & Google Sheets Structure

Google Apps Script & Spreadsheet Tabs Guide

Recommended spreadsheet setup for managing Student Data, Attendance Logs, and KLC Classes.

1 Required Spreadsheet Tabs & Column Headers

Make sure your Google Spreadsheet contains these exact sheet names and Row 1 column headers:

Tab 1: Student_Data

Row 1 Headers: ID Siswa | Full Name | Enrolled Classes | Class Name | Class ID | Status

Tab 2: Attendance_Log

Row 1 Headers: Timestamp | Date | Student ID | Student Name | Class ID | Class Name | Status | Topic | Teacher Note | Remark

Tab 3 (Optional): KLC_Classes

Row 1 Headers: Class ID | Class Name | Category | Schedule | Room

function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  
  try {
    var data = JSON.parse(e.postData.contents);
    var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
    
    // 1. Handle addStudent
    if (data.action === 'addStudent') {
      var sheet = spreadsheet.getSheetByName('Student_Data'); 
      sheet.appendRow([data.nis, data.name, data.enrolledClasses || '', data.className || '', data.classId || '', data.status || 'Active']); 
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    // 2. Handle addClass (New Class Schedule)
    if (data.action === 'addClass') {
      var classSheet = spreadsheet.getSheetByName('KLC_Classes');
      if (!classSheet) {
        classSheet = spreadsheet.insertSheet('KLC_Classes');
        classSheet.appendRow(['Class ID', 'Class Name', 'Category', 'Schedule', 'Room']);
      }
      classSheet.appendRow([data.classId, data.className, data.category, data.schedule, data.room]);
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    // 3. Handle deleteStudent
    if (data.action === 'deleteStudent') {
      var sheet = spreadsheet.getSheetByName('Student_Data');
      var rows = sheet.getDataRange().getValues();
      var targetId = String(data.nis || '').trim().toUpperCase();
      
      for (var i = rows.length - 1; i >= 1; i--) {
        if (String(rows[i][0]).trim().toUpperCase() === targetId) {
          sheet.deleteRow(i + 1);
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON);
    }
    
    // 4. Handle Attendance Log submission
    var logSheet = spreadsheet.getSheetByName("Attendance_Log");
    var records = data.records || data.data;
    
    if (Array.isArray(records)) {
      records.forEach(function(r) {
        logSheet.appendRow([
          r.timestamp,
          r.tanggal || r.date,
          r.idSiswa || r.studentId,
          r.namaSiswa || r.studentName,
          r.kodeKelas || r.classId,
          r.namaKelas || r.className,
          r.status,
          r.materi || r.topic,
          r.teacherNotes || r.teacherNote,
          r.remarks || r.remark
        ]);
      });
    }
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})).setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // Fetch Classes
  var classes = [];
  var classSheet = ss.getSheetByName("KLC_Classes");
  if (classSheet) {
    var cRows = classSheet.getDataRange().getValues();
    for (var c = 1; c < cRows.length; c++) {
      var cr = cRows[c];
      if (!cr[0]) continue;
      classes.push({
        id: cr[0],
        name: cr[1] || cr[0],
        category: cr[2] || (String(cr[0]).toUpperCase().includes('ENG') ? 'English' : 'Computer'),
        schedule: cr[3] || 'Scheduled Session',
        room: cr[4] || 'Community Hall Kaura Manggis'
      });
    }
  }

  // Fetch Students
  var studentSheet = ss.getSheetByName("Student_Data");
  var rows = studentSheet.getDataRange().getValues();
  var students = [];
  var classMap = {};
  
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    var idSiswa = row[0];
    if (!idSiswa) continue;
    
    var fullName = row[1];
    var enrolledClasses = row[2];
    var className = row[3];
    var classId = row[4];
    var status = row[5] || 'Active';
    
    students.push({
      id: idSiswa,
      name: fullName,
      fullName: fullName,
      enrolledClasses: enrolledClasses,
      className: className,
      classId: classId,
      classes: enrolledClasses ? enrolledClasses.split(',').map(function(c) { return c.trim(); }) : (classId ? [classId] : []),
      status: status
    });
    
    if (classId && !classMap[classId]) {
      classMap[classId] = {
        id: classId,
        name: className || classId,
        category: classId.toUpperCase().includes('ENG') ? 'English' : 'Computer',
        schedule: 'Scheduled Session',
        room: 'Community Hall Kaura Manggis'
      };
    }
  }

  if (classes.length === 0) {
    for (var key in classMap) {
      classes.push(classMap[key]);
    }
  }

  // Fetch Logs
  var logSheet = ss.getSheetByName("Attendance_Log");
  var logs = [];
  if (logSheet) {
    var logRows = logSheet.getDataRange().getValues();
    for (var j = 1; j < logRows.length; j++) {
      var lr = logRows[j];
      if (!lr[2]) continue;
      
      var rawDate = lr[1];
      var cleanDateStr = '';
      if (rawDate instanceof Date) {
        var y = rawDate.getFullYear();
        var m = String(rawDate.getMonth() + 1).padStart(2, '0');
        var d = String(rawDate.getDate()).padStart(2, '0');
        cleanDateStr = y + '-' + m + '-' + d;
      } else {
        var strVal = String(rawDate || '').trim();
        if (strVal.includes('T')) {
          cleanDateStr = strVal.split('T')[0];
        } else if (strVal.includes(' ')) {
          cleanDateStr = strVal.split(' ')[0];
        } else {
          cleanDateStr = strVal;
        }
      }

      logs.push({
        timestamp: lr[0],
        tanggal: cleanDateStr,
        idSiswa: lr[2],
        namaSiswa: lr[3],
        kodeKelas: lr[4],
        namaKelas: lr[5],
        status: lr[6],
        materi: lr[7],
        teacherNotes: lr[8],
        remarks: lr[9]
      });
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    students: students,
    classes: classes,
    logs: logs
  })).setMimeType(ContentService.MimeType.JSON);
}
